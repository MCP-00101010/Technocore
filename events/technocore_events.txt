namespace = technocore_dispatch

country_event = {
    id = technocore_dispatch.1
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = {
            limit = { has_origin = origin_technocore }

            save_event_target_as = technocore_dispatch_owner

            capital_scope = {
                set_name = "Remnant"

                solar_system = {
                    every_system_planet = {
                        limit = {
                            has_planet_flag = technocore_starting_colony
                            NOT = { is_colony = yes }
                        }

                        create_colony = {
                            owner = event_target:technocore_dispatch_owner
                            species = owner_main_species
                            ethos = owner
                        }

                        if = {
                            limit = { has_planet_flag = technocore_admin_segment }
                            set_name = "Technocore Administrative Segment"
                        }
                        if = {
                            limit = { has_planet_flag = technocore_research_segment_1 }
                            set_name = "Technocore Research Segment 01"
                            add_deposit = d_payback_habitat_research
                        }
                        if = {
                            limit = { has_planet_flag = technocore_trade_segment }
                            set_name = "Technocore Trade Segment 02"
                        }
                        if = {
                            limit = { has_planet_flag = technocore_research_segment_2 }
                            set_name = "Technocore Research Segment 02"
                            add_deposit = d_payback_habitat_research
                        }

                        clear_blockers = yes
                        generate_start_buildings_and_districts = yes
                        generate_start_pops = yes
                    }

                    every_system_planet = {
                        limit = { NOT = { has_planet_flag = technocore_megastructure_initialized } }

                        save_event_target_as = technocore_megastructure_body
                        solar_system = { save_event_target_as = technocore_megastructure_system }

                        event_target:technocore_dispatch_owner = {
                            country_event = { id = technocore_vanilla.1 }
                            country_event = { id = technocore_giga.1 }
                            country_event = { id = technocore_hyperquasaric.1 }
                            country_event = { id = technocore_stellar_manipulation.1 }
                            country_event = { id = technocore_sparble.1 }
                        }

                        set_planet_flag = technocore_megastructure_initialized
                    }

                    every_system_planet = {
                        limit = { has_planet_flag = technocore_molten_planet }

                        clear_deposits = yes
                        change_pc = pc_molten
                    }
                }
            }
        }
    }
}
