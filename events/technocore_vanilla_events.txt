namespace = technocore_vanilla

# Ownership rule:
# This file owns flags that originate in vanilla Stellaris.
# If another mod changes what a vanilla-owned flag should spawn, the compat branch
# still lives here because the flag itself is vanilla-owned.
# Example: has_dyson_sphere -> vanilla dyson sphere by default, Gigas variants when present.

country_event = {
    id = technocore_vanilla.1
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = {
            limit = {
                exists = event_target:technocore_megastructure_body
                exists = event_target:technocore_megastructure_system
            }

            if = {
                limit = { event_target:technocore_megastructure_body = { is_star = yes } }
                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_dyson_sphere } }

                    event_target:technocore_megastructure_system = {
                        if = {
                            limit = { event_target:technocore_megastructure_body = { giga_is_o_star_for_megas = yes } }
                            spawn_megastructure = {
                                type = "dyson_sphere_5_o_star"
                                owner = root
                                planet = event_target:technocore_megastructure_body
                            }
                        }
                        else_if = {
                            limit = { has_global_flag = giga_dyson_scaling }

                            if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_b_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_b_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_m_giant_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_m_giant_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_a_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_a_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_f_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_f_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_g_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_g_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_k_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_k_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else_if = {
                                limit = { event_target:technocore_megastructure_body = { giga_is_m_star_for_megas = yes } }
                                spawn_megastructure = {
                                    type = "dyson_sphere_5_m_star"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                            else = {
                                spawn_megastructure = {
                                    type = "dyson_sphere_5"
                                    owner = root
                                    planet = event_target:technocore_megastructure_body
                                }
                            }
                        }
                        else = {
                            spawn_megastructure = {
                                type = "dyson_sphere_5"
                                owner = root
                                planet = event_target:technocore_megastructure_body
                            }
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_dyson_swarm } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "dyson_swarm_3"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_matter_decompressor } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "matter_decompressor_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                # Vanilla notes:
                # - Dyson Sphere uses vanilla's own has_dyson_sphere flag and can scale by
                #   star class when Gigastructural Engineering is present.
                # - Dyson Swarm uses the custom Technocore flag has_dyson_swarm.
                #   Vanilla does not define star-class Dyson Swarm variants.
                # - Matter Decompressor uses the custom Technocore flag has_matter_decompressor.
            }

            if = {
                limit = { event_target:technocore_megastructure_body = { NOT = { is_star = yes } } }
                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_science_nexus } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "think_tank_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_sentry_array } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "spy_orb_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_mega_art_installation } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "mega_art_installation_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_strategic_coordination_center } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "strategic_coordination_center_3"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_interstellar_assembly } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "interstellar_assembly_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_arc_furnace } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "orbital_arc_furnace_4"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                if = {
                    limit = { event_target:technocore_megastructure_body = { has_planet_flag = has_grand_archive } }

                    event_target:technocore_megastructure_system = {
                        spawn_megastructure = {
                            type = "grand_archive_0"
                            owner = root
                            planet = event_target:technocore_megastructure_body
                        }
                    }
                }

                # Vanilla notes:
                # - Science Nexus uses the custom Technocore flag has_science_nexus.
                # - Sentry Array uses the custom Technocore flag has_sentry_array.
                # - Mega Art Installation uses the custom Technocore flag has_mega_art_installation.
                # - Strategic Coordination Center uses the custom Technocore flag
                #   has_strategic_coordination_center.
                # - Interstellar Assembly uses the custom Technocore flag has_interstellar_assembly.
                # - Arc Furnace uses vanilla's own has_arc_furnace flag.
                # - Grand Archive uses the custom Technocore flag has_grand_archive.
            }
        }
    }
}
